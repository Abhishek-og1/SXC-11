Preprocessor Directive

Code:

  #include <stdio.h>  // Preprocessor directive to include the standard I/O library

#define PI 3.14159   // Macro definition

int main() {
    float radius, area;

    printf("Enter the radius of the circle: ");
    scanf("%f", &radius);

    area = PI * radius * radius;  // Using the defined macro

    printf("The area of the circle is: %.2f\n", area);
    
    return 0;
}

Error and Types

Code:

  #include <stdio.h>  // Preprocessor directive

int main() {
    //  Syntax Error: Missing semicolon
    int a = 10, b = 5  
    int sum = a + b;  // Correct formula (but previous error stops compilation)

    //  Logical Error: Incorrect formula for multiplication
    int product = a - b;  // Expected a * b, but used subtraction instead

    //  Runtime Error: Division by zero
    int x = 10, y = 0;
    int result = x / y;  // Causes a runtime crash

    printf("Sum: %d\n", sum);
    printf("Product (incorrect): %d\n", product);
    printf("Result of division: %d\n", result); // Won't execute due to runtime error

    return 0;
}
